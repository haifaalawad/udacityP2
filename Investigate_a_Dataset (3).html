def countRows(column): #start funnction
    rows = df[column].str.cat(sep='|') #Saparate all the strings with a |
    rows = pd.Series(rows.split('|')) #Split all the colomns by the |
    count = rows.value_counts(ascending=False) #Big to small
    return count # end function
